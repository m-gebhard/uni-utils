    Class JsonObject<T>

    Represents a generic JSON object that can be serialized and deserialized using Unity's JsonUtility. The derived class is required to have the [Serializable] attribute.

    Parameters

    The JSON string to deserialize.
    Returns
    Type Description
    T

    An object of type T deserialized from the JSON string.
    Examples
    [Serializable]
public class PlayerData : JSONObject<PlayerData>
{
    public string playerName;
    public int score;
}

string json = "{\"playerName\":\"Alex\",\"score\":150}";
PlayerData data = new PlayerData().FromJson(json);
Debug.Log($"{data.playerName} scored {data.score}"); // Output: Alex scored 150

    ToJson()

    Serializes the current object to a JSON string.

    Declaration
    public string ToJson()
    Returns
    Type Description
    System.String

    A JSON string representation of the current object.
    Examples
    [Serializable]
public class PlayerData : JSONObject<PlayerData>
{
    public string playerName;
    public int score;
}

PlayerData data = new PlayerData { playerName = "Alex", score = 150 };
string json = data.ToJson();
Debug.Log(json); // Output: {"playerName":"Alex","score":150}
